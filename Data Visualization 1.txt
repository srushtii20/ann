1. Use the inbuilt dataset 'titanic'. The dataset contains 891 rows and contains information about 
the passengers who boarded the unfortunate Titanic ship. Use the Seaborn library to see if we 
can find any patterns in the data. 
2. Write a code to check how the price of the ticket (column name: 'fare') for each passenger

********************************************************************************************************************

import seaborn as sns
import matplotlib.pyplot as plt
import pandas as pd

# Load Titanic dataset
titanic = sns.load_dataset('titanic')

# Display first 5 rows
print("Dataset Preview:")
print(titanic.head())

# Dataset information
print("\nDataset Info:")
print(titanic.info())

# Statistical summary
print("\nDataset Description:")
print(titanic.describe())

# 1. Survival Count Plot
plt.figure(figsize=(6, 5))

sns.countplot(x='survived', data=titanic)

plt.title("Survival Count (0 = No, 1 = Yes)")
plt.xlabel("Survived")
plt.ylabel("Count")

plt.show()

# 2. Survival by Gender
plt.figure(figsize=(6, 5))

sns.countplot(x='survived', hue='sex', data=titanic)

plt.title("Survival by Gender")
plt.xlabel("Survived")
plt.ylabel("Count")

plt.show()

# 3. Survival by Passenger Class
plt.figure(figsize=(6, 5))

sns.countplot(x='survived', hue='class', data=titanic)

plt.title("Survival by Class")
plt.xlabel("Survived")
plt.ylabel("Count")

plt.show()

# 4. Age Distribution
plt.figure(figsize=(7, 5))

sns.histplot(titanic['age'], bins=30, kde=True)

plt.title("Age Distribution")
plt.xlabel("Age")
plt.ylabel("Frequency")

plt.show()

# 5. Fare Distribution
plt.figure(figsize=(7, 5))

sns.histplot(titanic['fare'], bins=30, kde=True)

plt.title("Fare Distribution")
plt.xlabel("Fare Price")
plt.ylabel("Number of Passengers")

plt.show()
