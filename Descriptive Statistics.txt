import pandas as pd

# Load dataset
df = pd.read_csv(r"C:\Users\HP\Downloads\StudentsPerformance.csv")

# Display first 5 rows
print("Dataset Preview:")
print(df.head())

# Group data by gender
grouped = df.groupby('gender')

# Calculate summary statistics
result = grouped[['math score', 'reading score', 'writing score']].agg(
    ['mean', 'median', 'min', 'max', 'std']
)

# Display grouped statistics
print("\nGrouped Summary Statistics:")
print(result)

import pandas as pd
import matplotlib.pyplot as plt

# Load Iris dataset
iris = pd.read_csv(r"C:\Users\HP\Downloads\iris.csv")

# Display dataset preview
print("Iris Dataset Preview:")
print(iris.head())

# Group dataset by species
grouped = iris.groupby('species')

# Display statistics for each species
for name, group in grouped:
    print("\nSpecies:", name)
    print(group.describe())

# Create boxplot
iris.boxplot(by='species', figsize=(8, 6))

# Add title
plt.title("Iris Dataset Boxplot")
plt.suptitle("")

# Display plot
plt.show()
plt.show() 
iris.hist(figsize=(8,6)) 
plt.suptitle("Distribution of Iris Features") 
plt.show()
