import tensorflow as tf
import matplotlib.pyplot as plt

# 1. Load Data
(x_train, y_train), (x_test, y_test) = tf.keras.datasets.mnist.load_data()

# 2. Preprocess Data
x_train = x_train / 255.0
x_test = x_test / 255.0

x_train = x_train.reshape(-1, 28, 28, 1)
x_test = x_test.reshape(-1, 28, 28, 1)

# 3. Build SIMPLE MODEL
model1 = tf.keras.Sequential([
    tf.keras.layers.Conv2D(
        32, (3, 3),
        activation='relu',
        input_shape=(28, 28, 1)
    ),

    tf.keras.layers.MaxPooling2D(2, 2),

    tf.keras.layers.Flatten(),

    tf.keras.layers.Dense(
        64,
        activation='relu'
    ),

    tf.keras.layers.Dense(
        10,
        activation='softmax'
    )
])

# Compile Base Model
model1.compile(
    optimizer='adam',
    loss='sparse_categorical_crossentropy',
    metrics=['accuracy']
)

# 4. Train Base Model
history1 = model1.fit(
    x_train,
    y_train,
    epochs=5,
    validation_data=(x_test, y_test)
)

# 5. Build IMPROVED MODEL (with Dropout)
model2 = tf.keras.Sequential([
    tf.keras.layers.Conv2D(
        32, (3, 3),
        activation='relu',
        input_shape=(28, 28, 1)
    ),

    tf.keras.layers.MaxPooling2D(2, 2),

    tf.keras.layers.Flatten(),

    tf.keras.layers.Dense(
        64,
        activation='relu'
    ),

    tf.keras.layers.Dropout(0.3),   # tuning

    tf.keras.layers.Dense(
        10,
        activation='softmax'
    )
])

# Compile Improved Model
model2.compile(
    optimizer='adam',
    loss='sparse_categorical_crossentropy',
    metrics=['accuracy']
)

# 6. Train Improved Model
history2 = model2.fit(
    x_train,
    y_train,
    epochs=5,
    validation_data=(x_test, y_test)
)

# 7. Plot Accuracy Graph
plt.plot(history1.history['accuracy'], label='Base Train')
plt.plot(history1.history['val_accuracy'], label='Base Test')

plt.plot(history2.history['accuracy'], label='Improved Train')
plt.plot(history2.history['val_accuracy'], label='Improved Test')

plt.title("Accuracy Comparison")
plt.xlabel("Epochs")
plt.ylabel("Accuracy")
plt.legend()
plt.show()

# 8. Plot Loss Graph
plt.plot(history1.history['loss'], label='Base Train Loss')
plt.plot(history1.history['val_loss'], label='Base Test Loss')

plt.plot(history2.history['loss'], label='Improved Train Loss')
plt.plot(history2.history['val_loss'], label='Improved Test Loss')

plt.title("Loss Comparison")
plt.xlabel("Epochs")
plt.ylabel("Loss")
plt.legend()
plt.show()
